    /**
     * Loop and print wcTeams in array
     * ... repeat until you reach zero  ...
     */
    public void printPoem() {
        //begin the poem
        System.out.println();
        System.out.println("2018 World Cup Quarter Finalists");

        // wcTeams (non-primitive) defined in constructor knows its length
        int teamsCount = wcTeams.length;
        for (int i = teamsCount; i >= 1; i/=2)  //loops through 2D array length backwards
        {

            //this print statement shows current count of wcTeams
            //  concatenation (+) of the loop variable and string to form a countdown message
            System.out.println(i + " wcTeams left...");

            //how many separate parts are there in a world cup teams world cup teams?
            for (int row = 0; row < teamsCount; row++) {  //cycles through "cells" of 2d array

                /*cycles through columns to print
                each world cup teams part by part, will eventually print entire column*/
                for (int col = 0; col < wcTeams[row].length; col++) {

                    // prints specific part of the world cup teams from the column
                    System.out.print(wcTeams[row][col] + " ");

                    //this is new line between separate parts
                    System.out.println();
                }

                //this new line gives separation between stanza of poem
                System.out.println();
            }

            //countdown for poem, decrementing teamsCount variable by 1
            teamsCount /= 2;
        }

        //out of all the loops, prints finishing messages
        System.out.println("France Wins the WC");
        System.out.println("0000000000000000000000000000000000");
        System.out.println("             THE END              ");
        System.out.println("  .-=========-.");
        System.out.println("  /'-=======-'/");
        System.out.println("  _|   .=.   |_");
        System.out.println(" ((|{{France}}|))");
        System.out.println("  /|   /|/   |/");
        System.out.println("  /_2018_WC__/");
        System.out.println("    _`) (`_   ");
        System.out.println("   _/_______/_");
        System.out.println(" //___________\\");
        
    }

    /**
    * A Java Driver/Test method that is the entry point for execution
    */
    public static void main(String[] args)  {
        new WcLoop().printPoem();   //a new world cup teams list and output in one step
    }
