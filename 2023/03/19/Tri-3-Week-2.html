 Objective would be to push more functionality into this Class to enforce consistent definition

public abstract class Bases {
	public final String masterType = "Base";
	private String type;	// extender should define their data type

	// generic enumerated interface
	public interface KeyTypes {
		String name();
	}
	protected abstract KeyTypes getKey();  	// this method helps force usage of KeyTypes

	// getter
	public String getMasterType() {
		return masterType;
	}

	// getter
	public String getType() {
		return type;
	}

	// setter
	public void setType(String type) {
		this.type = type;
	}
	
	// this method is used to establish key order
	public abstract String toString();

	// static print method used by extended classes
	public static void print(Bases[] objs) {
		// print 'Object' properties
		System.out.println(objs.getClass() + " " + objs.length);

		// print 'Bases' properties
		if (objs.length > 0) {
			Bases obj = objs[0];	// Look at properties of 1st element
			System.out.println(
					obj.getMasterType() + ": " + 
					obj.getType() +
					" listed by " +
					obj.getKey());
		}

		// print "Bases: Objects'
		for(Object o : objs)	// observe that type is Opaque
			System.out.println(o);

		System.out.println();
	}
}

public class Problems extends Bases {
	// Class data
	public static KeyTypes key = KeyType.title;  // static initializer
	public static void setOrder(KeyTypes key) {Problems.key = key;}
	public enum KeyType implements KeyTypes {title, difficulty, language, topic}

	// Instance data
	private final String language;
	private final String topic;
	private final int difficulty;

	// Constructor
	Problems(String language, String topic, int difficulty)
	{
		this.setType("Problems");
		this.language = language;
		this.topic = topic;
		this.difficulty = difficulty;
	}

	/* 'Bases' requires getKey to help enforce KeyTypes usage */
	@Override
	protected KeyTypes getKey() { return Problems.key; }

	/* 'Bases' requires toString override
	 * toString provides data based off of Static Key setting
	 */
	@Override
	public String toString() {		
		String output="";
		if (KeyType.difficulty.equals(this.getKey())) {
			output += this.difficulty;
		} else if (KeyType.language.equals(this.getKey())) {
			output += this.language;
		} else if (KeyType.topic.equals(this.getKey())) {
			output += "00" + this.topic;
			output = output.substring(output.length() - 2);
		} else {
			output = super.getType() + ": " + this.difficulty + ", " + this.language + ", " + this.topic;
		}
		return output;
	}

	// Test data initializer
	public static Problems[] cupcakes() {
		return new Problems[]{
				new Problems("Java", "String Implementation", 2),
				new Problems("Python", "Create Functions", 1),
				new Problems("C#", "Unity 3D", 4),
				new Problems("Java", "Classes", 3),
				new Problems("Java", "Inheritance", 3),
		};
	}
	
	public static void main(String[] args)
	{
		// Inheritance Hierarchy
		Problems[] objs = cupcakes();

		// print with title
		Problems.setOrder(KeyType.title);
		Problems.print(objs);

		// print difficulty only
		Problems.setOrder(KeyType.difficulty);
		Problems.print(objs);
	}
	
}

class Page {
     
    // Attributes of a complex number
    private double re, im;
     
    // Constructor to initialize a complex number
    // Default
    // public Page() {
    //     this.re = 0;
    //     this.im = 0;
    // }
     
    // Constructor 2: Parameterized
    public Page(double re, double im) {
         
        // This keyword refers to
        // current complex number
        this.re = re;
        this.im = im;
    }
    // Getters
    public double getReal() {
        return this.re;
    }
    public double getImaginary() {
        return this.im ;
    }
    // Setters
    public void setReal(double re) {
        this.re = re;
    }
    public void setImaginary(double im) {
        this.im = im;
    }
    // Overriding toString() method of String class
    @Override
    public String toString() {
        return this.re + " + " + this.im + "i";
    }
}

public class Articles {
        
    // the Articles class has
    // three fields
    public int header;
    public int body;
    public int footer;
        
    // the Articles class has
    // one constructor
    // lines for each part of the article
    public Articles(int startHeader, int startFooter, int startBody) {
        body = startBody;
        header = startHeader;
        footer = startFooter;
    }
        
    // the Articles class has
    // four methods
    public void setHeader(int newValue) {
        header = newValue;
    }
        
    public void setBody(int newValue) {
        body = newValue;
    }
        
    public void applyBrake(int decrement) {
        footer -= decrement;
    }
        
    public void footerUp(int increment) {
        footer += increment;
    }
        
}
